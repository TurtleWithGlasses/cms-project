Authentication

Two authentication methods are supported. Use JWT Bearer for user-facing integrations and API Keys for server-to-server / machine-to-machine communication.

Method Header How to obtain Expiry
JWT Bearer Authorization: Bearer <token> POST /auth/token 30 minutes (configurable)
API Key X-API-Key: <key> POST /api/v1/api-keys Set at creation (optional)

Obtaining a JWT token

# Obtain a JWT token
curl -s -X POST http://localhost:8000/auth/token \
  -H "Content-Type: application/x-www-form-urlencoded" \
  -d "username=you@example.com&password=secret" \
  | jq '.access_token'

# Use the token
curl http://localhost:8000/api/v1/content/ \
  -H "Authorization: Bearer <token>"
import httpx

BASE = "http://localhost:8000"

# Step 1 â€” authenticate
r = httpx.post(f"{BASE}/auth/token", data={
    "username": "you@example.com",
    "password": "secret",
})
r.raise_for_status()
token = r.json()["access_token"]

# Step 2 â€” make authenticated requests
headers = {"Authorization": f"Bearer {token}"}
content = httpx.get(f"{BASE}/api/v1/content/", headers=headers).json()
print(content)
const BASE = "http://localhost:8000";

// Step 1 â€” authenticate
const form = new URLSearchParams({
  username: "you@example.com",
  password: "secret",
});
const { access_token } = await fetch(`${BASE}/auth/token`, {
  method: "POST",
  body: form,
}).then(r => r.json());

// Step 2 â€” make authenticated requests
const content = await fetch(`${BASE}/api/v1/content/`, {
  headers: { Authorization: `Bearer ${access_token}` },
}).then(r => r.json());
console.log(content);

Using an API Key

# Create an API key (requires JWT auth)
curl -s -X POST http://localhost:8000/api/v1/api-keys \
  -H "Authorization: Bearer <token>" \
  -H "Content-Type: application/json" \
  -d '{"name": "My Integration", "expires_in_days": 90}'

# Use the API key
curl http://localhost:8000/api/v1/content/ \
  -H "X-API-Key: <your-api-key>"
import httpx

API_KEY = "your-api-key-here"
BASE = "http://localhost:8000"

client = httpx.Client(
    base_url=BASE,
    headers={"X-API-Key": API_KEY},
)
content = client.get("/api/v1/content/").json()
print(content)

Quickstart â€” Create & Publish Content

A complete example: authenticate, create a content item, and publish it.

# 1. Obtain a token
TOKEN=$(curl -s -X POST http://localhost:8000/auth/token \
  -d "username=admin@example.com&password=secret" | jq -r '.access_token')

# 2. Create a draft content item
CONTENT_ID=$(curl -s -X POST http://localhost:8000/api/v1/content/ \
  -H "Authorization: Bearer $TOKEN" \
  -H "Content-Type: application/json" \
  -d '{
    "title": "Hello World",
    "body": "My first article via the API.",
    "slug": "hello-world",
    "status": "draft"
  }' | jq '.id')

# 3. Publish it
curl -X POST http://localhost:8000/api/v1/content/$CONTENT_ID/publish \
  -H "Authorization: Bearer $TOKEN"
import httpx

BASE = "http://localhost:8000"
r = httpx.post(f"{BASE}/auth/token", data={"username": "admin@example.com", "password": "secret"})
token = r.json()["access_token"]
h = {"Authorization": f"Bearer {token}"}

# Create draft
content = httpx.post(f"{BASE}/api/v1/content/", headers=h, json={
    "title": "Hello World",
    "body": "My first article via the API.",
    "slug": "hello-world",
    "status": "draft",
}).json()

# Publish
httpx.post(f"{BASE}/api/v1/content/{content['id']}/publish", headers=h)
print("Published!")
const BASE = "http://localhost:8000";
const { access_token } = await fetch(`${BASE}/auth/token`, {
  method: "POST", body: new URLSearchParams({ username: "admin@example.com", password: "secret" }),
}).then(r => r.json());
const h = { Authorization: `Bearer ${access_token}`, "Content-Type": "application/json" };

// Create draft
const { id } = await fetch(`${BASE}/api/v1/content/`, {
  method: "POST", headers: h,
  body: JSON.stringify({ title: "Hello World", body: "My first article.", slug: "hello-world", status: "draft" }),
}).then(r => r.json());

// Publish
await fetch(`${BASE}/api/v1/content/${id}/publish`, { method: "POST", headers: h });
console.log("Published!");

GraphQL

The GraphQL API at /graphql supports both JWT and API Key authentication via request context.

# Fetch published content items
curl -s -X POST http://localhost:8000/graphql \
  -H "Authorization: Bearer <token>" \
  -H "Content-Type: application/json" \
  -d '{"query": "{ contents { id title slug status } }"}'
# Create a content item via GraphQL
curl -s -X POST http://localhost:8000/graphql \
  -H "Authorization: Bearer <token>" \
  -H "Content-Type: application/json" \
  -d '{"query": "mutation { createContent(title: \"Hello\", body: \"World\", slug: \"hello\") { id title } }"}'
